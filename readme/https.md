server{
    listen 80;
    listen [::]:80;
    root /var/www/backend/public;
    index index.php;
    server_name  backend.inhinyeru.com;
    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }
    location ~ \.php$ {
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/var/run/php/php8.0-fpm.sock;
    }
    location ~ /\.ht {
        deny all;
    }
}

nginx -t
